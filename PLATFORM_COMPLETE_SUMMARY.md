## 🔗 API Integration Examples

### JavaScript/React Example
```javascript
// API Client
const api = {
  baseURL: 'http://your-domain.com/api',
  token: localStorage.getItem('token'),
  
  async request(endpoint, options = {}) {
    const response = await fetch(`${this.baseURL}${endpoint}`, {
      ...options,
      headers: {
        'Authorization': `Bearer ${this.token}`,
        'Content-Type': 'application/json',
        ...options.headers
      }
    });
    return response.json();
  },
  
  // Products
  getProducts: (params) => api.request(`/catalog/products?${new URLSearchParams(params)}`),
  getProduct: (slug) => api.request(`/catalog/products/${slug}`),
  
  // Cart
  calculateCart: (items) => api.request('/cart/summary', {
    method: 'POST',
    body: JSON.stringify({ items })
  }),
  
  // Orders
  createOrder: (data) => api.request('/checkout/process', {
    method: 'POST',
    body: JSON.stringify(data)
  }),
  
  // Admin
  getDashboard: () => api.request('/admin/dashboard'),
  getOrders: (params) => api.request(`/admin/orders?${new URLSearchParams(params)}`),
};
```
